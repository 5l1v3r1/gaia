121 Status S3ReadFile::Open() {
122  string url = absl::StrCat("/", read_obj_url_);
123  h2::request<h2::empty_body> req{h2::verb::get, url, 11};
124 
125  if (offs_)
126  SetRange(offs_, kuint64max, &req);
127 
128  VLOG(1) << "Unsigned request: " << req;
129 
130  aws_.Sign(pool_->domain(), &req);
131 
132  // TODO: to wrap IO operations with retryable mechanism like in GCS.
133  HttpsClientPool::ClientHandle handle = pool_->GetHandle();
134 
135  system::error_code ec = handle->Send(req);
136 
137  if (ec) {
138  return ToStatus(ec);
139  }
140 
141  parser_.body_limit(kuint64max);
142  ec = handle->ReadHeader(&parser_);
143  if (ec) {
144  return ToStatus(ec);
145  }
146 
147  CHECK(parser_.keep_alive()) << "TBD";
148  const auto& msg = parser_.get();
149 
150  if (msg.result() != h2::status::ok) {
151  LOG(INFO) << "OpenError: " << msg;
152 
153  return Status(StatusCode::IO_ERROR, string(msg.reason()));
154  }
155  VLOG(1) << "HeaderResp(" << handle->native_handle() << "): " << msg;
156 
157  auto content_len_it = msg.find(h2::field::content_length);
158  if (content_len_it != msg.end()) {
159  size_t content_sz = 0;
160  CHECK(absl::SimpleAtoi(absl_sv(content_len_it->value()), &content_sz));
161 
162  if (size_) {
163  CHECK_EQ(size_, content_sz + offs_) << "File size has changed underneath during reopen";
164  } else {
165  size_ = content_sz;
166  }
167  }
168  https_handle_ = std::move(handle);
169  return Status::OK;
170 }

172 StatusObject<size_t> S3ReadFile::Read(size_t offset, const strings::MutableByteRange& range) {
173  CHECK(!range.empty());
174 
175  if (offset != offs_) {
176  return Status(StatusCode::INVALID_ARGUMENT, "Only sequential access supported");
177  }
178 
179  // We can not cache parser() into local var because Open() below recreates the parser instance.
180  if (parser_.is_done()) {
181  return 0;
182  }
183 
184  size_t read_sofar = 0;
185  while (read_sofar < range.size()) {
186  // We keep body references inside the loop because Open() that might be called here,
187  // will recreate the parser from the point the connections disconnected.
188  auto& body = parser()->get().body();
189  auto& left_available = body.size;
190  body.data = range.data() + read_sofar;
191  left_available = range.size() - read_sofar;
192 
193  error_code ec = https_handle_->Read(parser()); // decreases left_available.
194  size_t http_read = (range.size() - read_sofar) - left_available;
195 
196  if (!ec || ec == h2::error::need_buffer) { // Success
197  DVLOG(2) << "Read " << http_read << " bytes from " << offset << " with capacity "
198  << range.size() << "ec: " << ec;
199 
200  // This check does not happen. See here why: https://github.com/boostorg/beast/issues/1662
201  // DCHECK_EQ(sz_read, http_read) << " " << range.size() << "/" << left_available;
202  offs_ += http_read;
203 
204  CHECK(left_available == 0 || !ec);
205  return http_read + read_sofar;
206  }
207 
208  if (ec == h2::error::partial_message) {
209  offs_ += http_read;
210  VLOG(1) << "Got partial_message, socket status: "
211  << https_handle_->client()->next_layer().status() << ", socket "
212  << https_handle_->native_handle();
213 
214  // advance the destination buffer as well.
215  read_sofar += http_read;
216  ec = asio::ssl::error::stream_truncated;
217  }
218 
219  if (ec == asio::ssl::error::stream_truncated) {
220  VLOG(1) << "Stream " << read_obj_url_ << " truncated at " << offs_ << "/" << size_;
221  https_handle_.reset();
222 
223  RETURN_IF_ERROR(Open());
224  VLOG(1) << "Reopened the file, new size: " << size_;
225  // TODO: to validate that file version has not been changed between retries.
226  continue;
227  } else {
228  LOG(ERROR) << "ec: " << ec << "/" << ec.message() << " at " << offset << "/" << size_;
229  LOG(ERROR) << "FiberSocket status: " << https_handle_->client()->next_layer().status();
230 
231  return ToStatus(ec);
232  }
233  }
234 
235  return read_sofar;
236 }

306 ListS3BucketResult ListS3Buckets(const AWS& aws, http::HttpsClientPool* pool) {
307  HttpsClientPool::ClientHandle handle = pool->GetHandle();
308 
309  h2::request<h2::empty_body> req{h2::verb::get, "/", 11};
310  h2::response<h2::string_body> resp;
311 
312  aws.Sign(S3Bucket::kRootDomain, &req);
313 
314  system::error_code ec = handle->Send(req, &resp);
315 
316  if (ec) {
317  return ToStatus(ec);
318  }
319 
320  if (resp.result() != h2::status::ok) {
321  LOG(INFO) << "Error: " << resp;
322 
323  return Status(StatusCode::IO_ERROR, string(resp.reason()));
324  }
325 
326  VLOG(1) << "ListS3Buckets: " << resp;
327 
328  std::vector<std::string> res;
329 
330  xmlDocPtr doc = xmlReadMemory(resp.body().data(), resp.body().size(), NULL, NULL, 0);
331  CHECK(doc);
332 
333  xmlXPathContextPtr xpathCtx = xmlXPathNewContext(doc);
334 
335  auto register_res = xmlXPathRegisterNs(xpathCtx, BAD_CAST "NS",
336  BAD_CAST "http://s3.amazonaws.com/doc/2006-03-01/");
337  CHECK_EQ(register_res, 0);
338 
339  xmlXPathObjectPtr xpathObj = xmlXPathEvalExpression(
340  BAD_CAST "/NS:ListAllMyBucketsResult/NS:Buckets/NS:Bucket/NS:Name", xpathCtx);
341  CHECK(xpathObj);
342  xmlNodeSetPtr nodes = xpathObj->nodesetval;
343  if (nodes) {
344  int size = nodes->nodeNr;
345  for (int i = 0; i < size; ++i) {
346  xmlNodePtr cur = nodes->nodeTab[i];
347  CHECK_EQ(XML_ELEMENT_NODE, cur->type);
348  CHECK(cur->ns);
349  CHECK(nullptr == cur->content);
350 
351  if (cur->children && cur->last == cur->children && cur->children->type == XML_TEXT_NODE) {
352  CHECK(cur->children->content);
353  res.push_back(as_char(cur->children->content));
354  }
355  }
356  }
357 
358  xmlXPathFreeObject(xpathObj);
359  xmlXPathFreeContext(xpathCtx);
360  xmlFreeDoc(doc);
361 
362  return res;
363 }
